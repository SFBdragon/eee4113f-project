#include <Arduino.h>
#include "esp_wifi.h"
#include "nvs_flash.h"

void setup() {
    Serial.begin(115200);
    delay(1000); // give host time to attach monitor
    // NVS must be initialized before Wi-Fi
    nvs_flash_init();
    wifi_init_config_t cfg = WIFI_INIT_CONFIG_DEFAULT();
    esp_wifi_init(&cfg);
    esp_wifi_set_mode(WIFI_MODE_STA);
    esp_wifi_start(); // MAC is only valid after start
    uint8_t mac[6];
    esp_wifi_get_mac(WIFI_IF_STA, mac);
    Serial.printf("STA MAC: %02X:%02X:%02X:%02X:%02X:%02X\n",
                  mac[0], mac[1], mac[2], mac[3], mac[4], mac[5]);
    uint8_t base_mac[6];
    esp_efuse_mac_get_default(base_mac);
    Serial.printf("Base MAC (eFuse): %02X:%02X:%02X:%02X:%02X:%02X\n",
                  base_mac[0], base_mac[1], base_mac[2],
                  base_mac[3], base_mac[4], base_mac[5]);
}
void loop() {}
